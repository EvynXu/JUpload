Welcome to jupload.

This is an upload script library on mobile devices.Welcome everyone to use, as well as various proposals.

a single demo

new JUpload('#demo',{
postUrl: "your server path",
onClientLoad:function(e,file,fr){
    //can read local file in fr.result
},
onServerProgress:function(e){
    var totalSize = e.total;
    var current = e.loaded;
    $('#progress').html("onServerProgress:<br/>-TotalSize:" + totalSize + "<br/>-currentSize:" + current);
},
onSuccess: function (e, file, responseText) {
    var div = document.createElement('div');
    div.innerHTML = "returnText = " + responseText;
    document.body.appendChild(div);
}
